<script type="text/javascript">
var baseTitle = "KIT 中野研 在席状況";

function showStudents(data) {
var num_present = data.num_present;
var num_absent = data.num_absent;
var students = data.students;
document.title = "[" + num_present + "] " + baseTitle;
$("#tbl_body").empty();
if (num_present > 0 && num_absent == 0) {
$("#tbl_title").html('<button class="btn btn-warning btn-lg" type="button">満員御礼 <span class="badge">' + num_present + '</span></button>');
} else if (num_present > 0) {
$("#tbl_title").html('<button class="btn btn-success btn-lg" type="button">在席中 <span class="badge">' + num_present + '</span></button>');
} else {
$("#tbl_title").html('<button class="btn btn-danger btn-lg" type="button">ただいま無人</button>');
}
$.each(students, function(i, student) {
if (student.room_name == '') {
var trtag = "<tr>";
} else if (num_absent == 0) {
var trtag = "<tr class='warning'>";
} else {
var trtag = "<tr class='success'>";
}
var $tr = $(trtag).append($("<td>").text(student.class_name),
$("<td>").text(student.name),
$("<td>").text(student.room_name));
// console.log($tr.wrap('<p>').html());
$tr.appendTo("#tbl_body");
});
};

// Show students on page load
var jsonUrl = "/lab.json/nakano";
$.ajax({url: jsonUrl, success: showStudents, dataType: "json"});

// Update students every 30 seconds
(function poll(){
setTimeout(function() {
$.ajax({url: jsonUrl,
success: function(data) {
showStudents(data);
poll();
},
error: function(request, status, error) {
poll();
},
dataType: "json"});
}, 30000);
})();
</script>
